customer > add > gen_info > pname

Request:

<packet>
  <customer>
    <add>
      <gen_info>
        <pname>John Smith</pname>
        <login>john-unit-test</login>
        <passwd>simple-password</passwd>
      </gen_info>
    </add>
  </customer>
</packet>

Response:

<packet version="1.6.6.0">
  <customer>
    <add>
      <result>
        <status>ok</status>
        <id>4</id>
        <guid>b9f1aa18-baba-41b5-a6f2-04e457c16165</guid>
      </result>
    </add>
  </customer>
</packet>
