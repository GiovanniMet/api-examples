customer > add > gen_info > pname

Request:

<packet>
  <customer>
    <add>
      <gen_info>
        <pname>John Smith</pname>
        <login>john-unit-test</login>
        <passwd>simple-password</passwd>
      </gen_info>
    </add>
  </customer>
</packet>

Response:

<packet version="1.6.6.0">
  <customer>
    <add>
      <result>
        <status>ok</status>
        <id>42</id>
        <guid>2c54b044-c449-4f8d-908b-31f027d84287</guid>
      </result>
    </add>
  </customer>
</packet>
